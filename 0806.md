### 4837 부분집합의 합

```python
T = int(input())

for h in range(1, T + 1):
    n, k = map(int, input().split())
    A = [i for i in range(1, 13)]
    b = len(A)
    tc_li = []
    for i in range(1 << b):
        total = 0
        cnt = 0
        for j in range(b + 1):
            if i & (1 << j):
                total += A[j]
                cnt += 1
        tc_li += [[total, cnt]]



    count = 0
    for i in range(len(tc_li)):
        if tc_li[i][0] == k and tc_li[i][1] == n:
            count += 1

    print('#%d %d' %(h, count))

```

### 4839 이진탐색

```python
import sys
sys.stdin = open('sample_input.txt', 'r')

T = int(input())

for i in range(1, T+1):
    P, A, B = map(int,input().split())
    l = 1
    r = P
    c = int((l + r) // 2)
    count = 0
    while True:

        if c == A:
            count += 1
            break
        elif c < A :
            l = c
            c = int((l+r)//2)
            count += 1
            continue
        else:
            r = c
            c = int((l+r)//2)
            count += 1
            continue

    l = 1
    r = P
    c = int((l + r) // 2)
    count_b = 0
    while True:

        if c == B:
            count_b += 1
            break
        elif c < B :
            l = c
            c = int((l+r)//2)
            count_b += 1
            continue
        else:
            r = c
            c = int((l+r)//2)
            count_b += 1
            continue

    if count == count_b :
        print('#%d 0'%(i))
    elif count < count_b :
        print('#%d A' %(i))
    else :
        print('#%d B' %(i))

```

#import pprint

T = int(input())

for i in range(1, T+1):
    N, M = map(int,input().split())
    m_li = []
    for _ in range(N):
        m_li.append(list(map(int,input().split())))
    #pprint.pprint(m_li)
    max_total = 0

    for _ in range(N-M+1):
        for j in range(N-M+1):
            total = 0
            for a in range(_, _+M):
                for b in range(j, j+M):
                    total += m_li[a][b]
            if max_total < total:
                max_total = total

    print('#%d %d' %(i, max_total))
