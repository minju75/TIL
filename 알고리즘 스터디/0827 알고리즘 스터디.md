def check_bracket(target):
    stack = list()
    for i in range(len(target)):
        if target[i] == '(' or target[i] == "[" or target[i] == "{":
            stack.append(target[i])
        if target[i] == ')' or target[i] == "]" or target[i] == "}":
            if len(stack) == 0:
                return 0
            tmp = stack.pop()
            if target[i] == ")" and tmp == "(":
                continue
            if target[i] == "]" and tmp == "[":
                continue
            if target[i] == "}" and tmp == "{":
                continue
            return 0

    if len(stack) != 0:
        return 0
    else:
        return 1

    for j in n:
        ta[j[0]][j[1]] = 1
    q, e = map(int,input().split())
    m = []
    while True :
        if 1 in ta[q]:

            ta[q][k] = 0

            q = k
